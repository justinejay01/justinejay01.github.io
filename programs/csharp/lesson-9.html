Performance Task for Lesson 9

Create 2 Classes, Class “Program” containing the Main Method and Class “Compute” with the ff methods ( Add, Subtract, Multiply , and Divide ). In the Main method , ask the user to input 2 numbers (num1 and num2) and then ask the user to select one from the options (+, - , *, /, #). If the user entered ‘#’ , then the program terminates. The program will then perform the task.

Note: Declare an object named “com” for your class Compute inside the Main Method. (Use do- while loop, switch case, Classes and Object).

class Compute
{
 public void Add(double x, double y)
 {
  Console.WriteLine("{0} {1} {2} = {3}", x, "+", y, x + y);
 }
 public void Subtract(double x, double y)
 {
  Console.WriteLine("{0} {1} {2} = {3}", x, "-", y, x - y);
 }
 public void Multiply(double x, double y)
 {
  Console.WriteLine("{0} {1} {2} = {3}", x, "*", y, x * y);
 }
 public void Divide(double x, double y)
 {
  Console.WriteLine("{0} {1} {2} = {3}", x, "/", y, x / y);
 }
}

class Program
{
 static void Main(string[] args)
 {
  do
  {
   double num1, num2;
   char op;
   Console.Write("Enter value for num1: ");
   num1 = Convert.ToDouble(Console.ReadLine());
   Console.Write("Enter value for num2: ");
   num2 = Convert.ToDouble(Console.ReadLine());
   Console.WriteLine("------OPERATORS------");
   Console.WriteLine("[+] ADD");
   Console.WriteLine("[-] SUBTRACT");
   Console.WriteLine("[*] MULTIPLY");
   Console.WriteLine("[/] DIVIDE");
   Console.WriteLine("[#] EXIT");
   Console.Write("Enter your choice: ");
   op = Convert.ToChar(Console.ReadLine());

   if (op != '#')
   {
    Compute com = new Compute();
    switch (op)
    {
     case '+':
     com.Add(num1, num2);
     break;
     case '-':
     com.Subtract(num1, num2);
     break;
     case '*':
     com.Multiply(num1, num2);
     break;
     case '/':
     com.Divide(num1, num2);
     break;
     default:
     Console.WriteLine("Invalid input!");
     break;
    }
   continue;
   }
   else
   {
    Console.Write("End Program");
    Console.ReadLine();
    break;
   }
  } while (true);
 }
}
